107 void calc_mu(MatDoub &y)
108 {
109 const size_t n = y.cols();
110
111 gsl_interp_accel *acc = gsl_interp_accel_alloc();
112 gsl_spline *spline = gsl_spline_alloc(gsl_interp_cspline, n);
113
114 std::vector<double> zz, phigrad;
115 for (size_t k = 0; k < y.cols(); k++)
116 {
117 zz.push_back(z[k]);
118 double t = 0;
119 for (size_t j = 0; j < dim; j++)
120 t += pow(y[j][k], 2);
121 phigrad.push_back(t);
122 }
123
124 gsl_spline_init(spline, zz.data(), phigrad.data(), n);
125 double EnergyDissipated =
126 gsl_spline_eval_integ(spline, zz.front(), zz.back(), acc);
127
128 gsl_spline_free(spline);
129 gsl_interp_accel_free(acc);
130
131 mu = DeltaV / EnergyDissipated;
132 }

134 void smatrix(const int k,
135 const int k1,
136 const int k2,
137 const int jsf,
138 VecInt &indexv,
139 MatDoub &s,
140 MatDoub &y)
141 {
142 s.zero(); // Set matrix s = 0
143 if (k == k1)
144 {
145 // Calculate mu before anything else
146 calc_mu(y);
147 if (mode == ProfileSolverMode::Deriv)
148 {
149 // Boundary conditions dv/dz = 0 on first boundary
150 for (size_t field = 0; field < dim; field++)
151 {
152 // Sn at the first boundary
153 s[dim + field][2 * dim + field] = 1.0;
154 // B0
155 s[dim + field][jsf] = y[field][0];
156 }
157 }
158 else if (mode == ProfileSolverMode::Field)
159 {
160 for (size_t field = 0; field < dim; field++)
161 {
162 // Sn at the first boundary
163 s[dim + field][3 * dim + field] = 1.0;
164 // B0
165 s[dim + field][jsf] = y[dim + field][0] - TrueVacuum[field];
166 }
167 }
168 }
169 else if (k > k2 - 1)
170 {
171 if (mode == ProfileSolverMode::Deriv)
172 {
173 // Boundary conditionsd dv/dz = 0 on second boundary
174 for (size_t field = 0; field < dim; field++)
175 {
176 // Sn at the last boundary
177 s[field][2 * dim + field] = 1.0;
178 // C0
179 s[field][jsf] = y[field][k2 - 1];
180 }
181 }
182 else if (mode == ProfileSolverMode::Field)
183 {
184 // Fix the fields at |z| -> Infinity
185 for (size_t field = 0; field < dim; field++)
186 {
187 // Sn at the last boundary
188 s[field][3 * dim + field] = 1.0;
189 // C0
190 s[field][jsf] = y[dim + field][k2 - 1] - FalseVacuum[field];
191 }
192 }
193 }
194 else
195 {
196 const double dz = z[k] - z[k - 1];
197 std::vector<double> point; // middle point (phi(k) + phi(k-1)) / 2
198 for (size_t j = 0; j < dim; j++)
199 point.push_back((y[dim + j][k] + y[dim + j][k - 1]) / 2);
200 const std::vector<double> dv = dV(point);
201 const std::vector<std::vector<double>> hessian = Hessian(point);
202
203 // 0 <= j < dim -> phi'(z)
204 for (size_t j = 0; j < dim; j++)
205 {
206 for (size_t n = 0; n < dim; n++)
207 {
208 s[j][0 * dim + n] = -Delta(j, n) * (1 + dz * mu / 2); // 1
209 s[j][1 * dim + n] = -1. / 2. * dz * hessian[j][n]; // 3
210 s[j][2 * dim + n] = Delta(j, n) * (1 - dz * mu / 2); // 5
211 s[j][3 * dim + n] = -1. / 2. * dz * hessian[j][n]; // 7
212 }
213 // Equations for E(k,k-1)
214 s[j][jsf] = y[j][k] - y[j][k - 1] -
215 dz * (dv[j] - mu * (y[j][k] + y[j][k - 1]) / 2);
216 }
217
218 // dim <= j < 2 * dim -> phi(z)
219 for (size_t j = 0; j < dim; j++)
220 {
221 for (size_t n = 0; n < dim; n++)
222 {
223 s[dim + j][0 * dim + n] = -dz / 2. * Delta(j, n); // 2
224 s[dim + j][1 * dim + n] = -Delta(j, n); // 4
225 s[dim + j][2 * dim + n] = -dz / 2. * Delta(j, n); // 6
226 s[dim + j][3 * dim + n] = Delta(j, n); // 8
227 }
228 // Equations for E(k,k-1)
229 s[dim + j][jsf] = y[dim + j][k] - y[dim + j][k - 1] -
230 dz * (y[j][k] + y[j][k - 1]) / 2;
231 }
232 }
233 // Reorder columns
234 MatDoub ss = s;
235 for (size_t j = 0; j < 2 * dim; j++)
236 {
237 for (size_t n = 0; n < 2 * dim; n++)
238 {
239 s[j][n] = ss[j][indexv[n]];
240 s[j][2 * dim + n] = ss[j][2 * dim + indexv[n]];
241 }
242 }
243 }
