76 void smatrix(const int k,
77 const int k1,
78 const int k2,
79 const int jsf,
80 VecInt &indexv,
81 MatDoub &s,
82 MatDoub &y)
83 {
84 s.zero(); // Set matrix s = 0
85 if (k == k1)
86 {
87 if (mode == ProfileSolverMode::Deriv)
88 {
89 // Boundary conditions dv/dz = 0 on first boundary
90 for (size_t field = 0; field < dim; field++)
91 {
92 // Sn at the first boundary
93 s[dim + field][2 * dim + 1 + field] = 1.0;
94 // B0
95 s[dim + field][jsf] = y[field][0];
96 }
97 // Sn at the first boundary
98 s[dim + dim][3 * dim + 1] = 1.0;
99 // B0
100 s[dim + dim][jsf] = y[dim][0] - TrueVacuum[0];
101 }
102 else if (mode == ProfileSolverMode::Field)
103 {
104 for (size_t field = 0; field < dim; field++)
105 {
106 // Sn at the first boundary
107 s[dim + field][2 * dim + 1 + field] = 1.0;
108 // B0
109 s[dim + field][jsf] = y[dim + field][0] - TrueVacuum[field];
110 }
111 // Sn at the first boundary
112 s[dim + dim][3 * dim + 1] = 1.0;
113 // B0
114 s[dim + dim][jsf] = y[0][0];
115 }
116 }
117 else if (k > k2 - 1)
118 {
119 if (mode == ProfileSolverMode::Deriv)
120 {
121 // Boundary conditionsd dv/dz = 0 on second boundary
122 for (size_t field = 0; field < dim; field++)
123 {
124 // Sn at the last boundary
125 s[field][2 * dim + 1 + field] = 1.0;
126 // C0
127 s[field][jsf] = y[field][k2 - 1];
128 }
129 }
130 else if (mode == ProfileSolverMode::Field)
131 {
132 // Fix the fields at |z| -> Infinity
133 for (size_t field = 0; field < dim; field++)
134 {
135 // Sn at the last boundary
136 s[field][3 * dim + 1 + field] = 1.0;
137 // C0
138 s[field][jsf] = y[dim + field][k2 - 1] - FalseVacuum[field];
139 }
140 }
141 }
142 else
143 {
144 const double dz = z[k] - z[k - 1];
145 std::vector<double> point; // middle point (phi(k) + phi(k-1)) / 2
146 for (size_t j = 0; j < dim; j++)
147 point.push_back((y[dim + j][k] + y[dim + j][k - 1]) / 2);
148 const std::vector<double> dv = dV(point);
149 const std::vector<std::vector<double>> hessian = Hessian(point);
150
151 // 0 <= j < dim -> phi'(z)
152 for (size_t j = 0; j < dim; j++)
153 {
154 for (size_t n = 0; n < dim; n++)
155 {
156 s[j][0 * dim + n] =
157 -Delta(j, n) * (1 - dz / 2 * y[2 * dim][k - 1]); // 1
158 s[j][1 * dim + n] = -1. / 2. * dz * hessian[j][n]; // 3
159 s[j][2 * dim + n + 1] =
160 Delta(j, n) * (1 + dz / 2 * y[2 * dim][k]); // 5
161 s[j][3 * dim + n + 1] = -1. / 2. * dz * hessian[j][n]; // 7
162 }
163 s[j][2 * dim] = dz / 2. * y[j][k - 1]; // 11
164 s[j][4 * dim + 1] = dz / 2. * y[j][k]; // 16
165 // Equations for E(k,k-1)
166 s[j][jsf] = y[j][k] - y[j][k - 1] - dz * dv[j];
167 }
168
169 // dim <= j < 2 * dim -> phi(z)
170 for (size_t j = 0; j < dim; j++)
171 {
172 for (size_t n = 0; n < dim; n++)
173 {
174 s[dim + j][0 * dim + n] = -dz / 2. * Delta(j, n); // 2
175 s[dim + j][1 * dim + n] = -Delta(j, n); // 4
176 s[dim + j][2 * dim + n + 1] = -dz / 2. * Delta(j, n); // 6
177 s[dim + j][3 * dim + n + 1] = Delta(j, n); // 8
178 }
179 s[dim + j][2 * dim] = 0; // 12
180 s[dim + j][4 * dim + 1] = 0; // 17
181 // Equations for E(k,k-1)
182 s[dim + j][jsf] = y[dim + j][k] - y[dim + j][k - 1] -
183 dz * (y[j][k] + y[j][k - 1]) / 2;
184 }
185 // j = 2 * dim -> eta
186 for (size_t n = 0; n < dim; n++)
187 {
188 s[2 * dim][0 * dim + n] = 0; // 9
189 s[2 * dim][1 * dim + n] = 0; // 10
190 s[2 * dim][2 * dim + 1 + n] = 0; // 14
191 s[2 * dim][3 * dim + 1 + n] = 0; // 15
192 }
193 s[2 * dim][2 * dim] = -1; // 13
194 s[2 * dim][4 * dim + 1] = 1; // 18
195 // Equations for E(k,k-1) for eta
196 s[2 * dim][jsf] = y[2 * dim][k] - y[2 * dim][k - 1];
197 MatDoub ss = s;
198 int neq = 2 * dim + 1;
199 for (size_t j = 0; j < neq; j++)
200 {
201 for (size_t n = 0; n < neq; n++)
202 {
203 s[j][n] = ss[j][indexv[n]];
204 s[j][neq + n] = ss[j][neq + indexv[n]];
205 }
206 }
207 }
208 }
209};
