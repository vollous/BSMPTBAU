42template <typename key, typename value>
43std::unordered_map<value, key>
44InvertMap(const std::unordered_map<key, value> &originalMap,
45 const std::string &errorOnDuplicateValue)
46{
47 std::unordered_map<value, key> result;
48 for (const auto &[orig_key, orig_value] : originalMap)
49 {
50 auto success = result.emplace(orig_value, orig_key);
51 if (not success.second)
52 {
53 throw std::runtime_error(errorOnDuplicateValue);
54 }
55 }
56
57 return result;
58}

274template <typename T>
275std::vector<std::vector<T>> operator*(const std::vector<std::vector<T>> &A,
276 const std::vector<std::vector<T>> &B)
277{
278 if (A.empty() || B.empty())
279 {
280 throw std::runtime_error(
281 "Matrix multiplication cannot be done with empty matrices.");
282 }
283
284 const std::size_t m = A.size();
285 const std::size_t n = A[0].size();
286 const std::size_t p = B[0].size();
287
288 // Check A is rectangular
289 for (const auto &row : A)
290 {
291 if (row.size() != n)
292 {
293 throw std::runtime_error("Matrix A is not rectangular.");
294 }
295 }
296
297 // Check B is rectangular
298 for (const auto &row : B)
299 {
300 if (row.size() != p)
301 {
302 throw std::runtime_error("Matrix B is not rectangular.");
303 }
304 }
305
306 // Compatibility check
307 if (B.size() != n)
308 {
309 throw std::runtime_error(
310 "Matrix multiplication cannot be done. "
311 "Number of columns of A must equal number of rows of B.");
312 }
313
314 // Result matrix initialized with zeros
315 std::vector<std::vector<T>> C(m, std::vector<T>(p, T{}));
316
317 // Standard triple-loop multiplication
318 for (std::size_t i = 0; i < m; ++i)
319 {
320 for (std::size_t k = 0; k < n; ++k)
321 {
322 for (std::size_t j = 0; j < p; ++j)
323 {
324 C[i][j] += A[i][k] * B[k][j];
325 }
326 }
327 }
328
329 return C;
330}
